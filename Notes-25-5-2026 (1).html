import ipywidgets as widgets
from IPython.display import display, HTML
import random
from collections import deque
import time

# Configuration and Custom Logic for Distribution
HISTORY_SIZE = 10
number_history = deque(maxlen=HISTORY_SIZE)

# Queue to manage the specific distribution: 3(100k), 4(1M), 3(10M)
pool = []

def refill_pool():
    global pool
    # 3 items in 100,000s
    hundred_thousands = [random.randint(100000, 999999) for _ in range(3)]
    # 4 items in 1,000,000s
    one_millions = [random.randint(1000000, 9999999) for _ in range(4)]
    # 3 items in 10,000,000s
    ten_millions = [random.randint(10000000, 99999999) for _ in range(3)]
    
    pool = hundred_thousands + one_millions + ten_millions
    random.shuffle(pool)

refill_pool()

# Improved CSS and Global JS function
setup_html = """


"""
display(HTML(setup_html))

# Widgets
current_num_html = widgets.HTML(value='
000,000
')
history_grid_html = widgets.HTML(value='')
generate_button = widgets.Button(description="✨ START ✨")
generate_button.add_class("btn-cute")
js_trigger = widgets.HTML(value="")

def update_ui():
    current = number_history[0] if number_history else 0
    current_num_html.value = f'
{current:,}
'
    # Reverse items to show the correct count #1-#10 within the current round
    round_count = len(number_history)
    items_html = "".join([f'
#{round_count-i}
{num:,}
' for i, num in enumerate(number_history)])
    history_grid_html.value = f'
{items_html}
'
    if number_history:
        js_trigger.value = f""

def on_click(b):
    global pool
    # If we already have 10 numbers, clear history before starting new round
    if len(number_history) >= HISTORY_SIZE:
        number_history.clear()
        
    if not pool:
        refill_pool()
        
    new_number = pool.pop()
    number_history.appendleft(new_number)
    update_ui()

generate_button.on_click(on_click)

ui_content = widgets.VBox([
    widgets.HTML('
🌸 สุ่มตัวเลขตามเงื่อนไข (รอบละ 10) 🌸
'),
    widgets.Box([generate_button], layout=widgets.Layout(justify_content='center', margin='10px 0')),
    current_num_html,
    widgets.HTML('
'),
    widgets.HTML('ประวัติรอบปัจจุบัน (ครบ 10 ครั้งจะเริ่มใหม่):'),
    history_grid_html,
    js_trigger
], layout=widgets.Layout(align_items='stretch'))

main_container = widgets.Box([ui_content], layout=widgets.Layout(width='100%', justify_content='center'))
main_container.add_class("main-wrapper")

display(main_container)
update_ui()
